\begin{recipe}[
	bakingtime={\unit[90]{minutes}},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[350]{\faren}}},
	source=\href{https://www.today.com/recipes/make-patti-labelles-sweet-potato-pie-recipe-home-t56966}{Patti LaBelle}
	]{Sweet Potato Pie}\index{Pie}\index{Sweet Potato}\index{Thanksgiving}
\ingredients[11]{%
	\textbf{Pie Crust:} & \\
	\unit[1 \nicefrac{1}{2}]{cups} & all-purpose flour \\
	\unit[\nicefrac{1}{2}]{cup} & cold butter \\
	\unit[\nicefrac{1}{2}]{tsp} & salt \\
	\unit[\nicefrac{1}{3}]{cup} & ice water \\
	\textbf{Pie Filling:} & \\
	3 & large sweet potatoes \\
	\unit[\nicefrac{1}{2}]{cup} & melted butter \\
	\unit[150]{g} & light brown sugar \\
	\unit[100]{g} & granulated sugar \\
	2 & eggs \\
	\unit[\nicefrac{1}{4}]{cup} & half-and-half \\
	\unit[\nicefrac{3}{4}]{tsp} & ground cinnamon \\
	\unit[1]{tsp} & nutmeg \\
}
\preparation{%
	\step Sift the flour and salt into a medium bowl.
	\step Add the butter. Cut the butter into the flour until the mixture resembles coarse crumbs with a few pea sized bits.
	\step Gradually add enough of the water until the mixture clumps together.
	\step Gather up the dough and press into a thick disk. Wrap the dough and refrigerate for at least an hour or overnight.
	\step On a lightly floured surface, roll the dough into a 13" circle about \nicefrac{1}{8}-inch thick.
	\step Transfer to a 9-inch pie pan and trim with scissors to a 1-inch overhang. Then fold the dough into a crimp. Place in fridge until filling is almost ready.
	\step Bring a large pot of lightly salted water to a boil.
	\step Add the sweet potatoes and reduce the hear to medium. Cook for about 30 minutes or until the sweet potatoes are fork tender.
	\step Drain the potatoes and wait until they are cool enough to handle (you can also run them under cold water).
	\step Mash with until very smooth and measure out 3 cups for the filling.
	\step Preheat the oven to 400\faren.
	\step Remove the pie crust from the fridge, brush the interior with some of the melted butter.
	\step Sprinkle \nicefrac{1}{4} cup of the brown sugar over the bottom of the pie shell.
	\step Bake until the pie dough is set and just beginning to brown, about 15 minutes. Make sure to place a cookie sheet below to collect melted butter.
	\step Using the stand mixer, combine the mashed sweet potatoes, melted butter, brown sugar, granulated sugar, eggs, half-and-half, cinnamon, and nutmeg.
	\step Spread into the partially baked pie shell, smooth the top.
	\step Reduce the oven to 350\faren.
	\step Bake until a knife inserted in the center comes out clean, about 1.5 hours.
	\step Cool completely on a wire rack.
}
\end{recipe}{}
