\begin{recipe}[
	preparationtime={\unit[10]{min}},
	bakingtime={\unit[25-30]{min}},
	portion=\portion{16 brownies},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[350]{\faren}}},
	source=\href{https://cooking.nytimes.com/recipes/1013155-fudge-brownies}{New York Times Cooking}
	]{Fudge Brownies}\index{Chocolate}\index{Vegetarian}
\ingredients[10]{%
	\unit[]{} & \\
	\unit[1\nicefrac{1}{2}]{cups} & butter \\
	\unit[1\nicefrac{1}{2}]{cups} & walnut halves \\
	\unit[9]{oz} & roughly chopped chocolate \\
	3 & large eggs \\
	\unit[1]{tsp} & salt \\
	\unit[400]{grams} & granulated sugar \\
	\unit[1]{tbsp} & vanilla extract \\
	\unit[1\nicefrac{1}{2}]{cups} & all-purpose flour \\
}
\preparation{
	\step Grease a 9x9" glass baking pan.
	\step If using walnuts, spread on a bking sheet and toast in the oven at \unit[350]{\faren} until fragrant and lightly browned, 8-10 minutes. Let cool, then roughly chop.
	\step Melt the butter in a medium saucepan over low heat. Remove from heat and add the chocolate and cover until melted, about 10 minutes.
	\step In a large bowl, whisk together the eggs, salt, sugar and vanilla extract until thick, creamy and beginning to lighten. 
	\step Whisk the chocolate mixture until smooth then add to the egg mixture just until well combined. 
	\step Fold in the flour until it dissapears, using as few strokes as possible. Fold in the walnuts.
	\step Pour the batter into the prepared pan and bake until the tip of a knife inserted into the center comes out with a few moist crumbs, 25-30 minutes. 
	\step Allow to cool to room temperature then cover and leave in the pan for several hours or overnight before cutting into pan.
}

\end{recipe}
