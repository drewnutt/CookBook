\begin{recipe}[
	preparationtime={\unit[1]{hrs}},
	portion=\portion{8 servings},
	]{Vegetarian Chili}\index{Vegetarian}
\ingredients[]{%
	1 & large onion \\
	\unit[6-8]{stalks} & celery\\
	2 & bell peppers \\
	8-10 & tomatoes \\
	\unit[2]{cans} & beans \\
	\unit[\nicefrac{1}{2}]{lb} & "ground meat" crumbles \\
	\unit[3] {tbsp} & chili powder \\
	\unit[1\nicefrac{1}{2}]{tbsp} & cumin \\
	\unit[2]{tsp} & paprika \\
	\unit[2]{tsp} & garlic powder \\
	\unit[1]{tsp} & cayenne powder \\
	\unit[2]{tsp} & salt \\
}
\preparation{
	\step Dice the onion and cut cellery into \nicefrac{1}{2}" slices.
	\step Turn an instant pot onto high saute, add \nicefrac{1} tbsp of oil and saute the onion until golden, about 5 minutes.
	\step Add the "ground meat" crumbles and saute for another 5 minutes.
	\step Turn off the instant pot and add the celery, bell peppers, tomatoes. Stir to combine.
	\step Drain the beans and add to the instant pot.
	\step Add the spices and stir to combine.
	\step Pressure cook for 45 minutes. Let the pressure release naturally for 20 minutes.


}

\end{recipe}
