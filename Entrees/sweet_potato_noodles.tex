\begin{recipe}[
	preparationtime={\unit[]{}},
	portion=\portion{3-4},
	source=``dandumbrell (IG)''
	]{Sweet Potato Scissor-Cut Noodles}\index{Sweet Potato}
\ingredients[]{%
	\unit[500]{g} & sweet potatoes \\
	\unit[400]{g} & all-purpose flour \\
	\unit[\nicefrac{1}{2}]{tsp} & salt\\
	\unit[2]{tbsp} & white miso \\
	\unit[100]{g} & butter \\
	\unit[\nicefrac{3}{4}]{cup} & reserved pasta water \\
}
\preparation{
	\step Cut the sweet potatoes into large chunks and steam in an instant pot until fork tender.
	\step Add the sweet potatoes to a large bowl and mash until smooth. Add the flour and salt and mix until a dough forms.
	\step Knead the dough for 6-8 minutes until smooth. If needed to reduce stickiness, add more flour 20g at a time.
	\step Make the sauce by heating the butter in a large frying pan (big enough to hold the noodles) until browned, stirring frequently. Add the miso and stir until combined. Set aside.
	\step Bring a large pot of salted water to a very light simmer and using a oiled pair of scissors snip small pieces of dough from the dough ball directly into the water.
	\step Once all the dough has been cut, bring the water to a boil and cook for about 3 minutes until the noodles float to the top.
	\step Add the cooked noodles to directly to the sauce along with \nicefrac{3}{4} cup of the pasta water and toss to combine. Add more pasta water if needed to loosen the sauce.
	\step Serve immediately with a sprinkle of furikake.
}
