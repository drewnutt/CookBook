\begin{recipe}[
	preparationtime={\unit[1\nicefrac{1}{2}]{hours}},
	bakingtime={\unit[30]{min}},
	portion=\portion{6},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[375]{\faren}}},
	source=\href{https://cooking.nytimes.com/recipes/1019221-vegetarian-shepherds-pie}{NYT Cooking}
	]{Shepherd's Pie}\index{Cast Iron}\index{Casserole}
\ingredients[20]{%
	\unit[2]{tbsp} & salt \\
	\unit[2]{lbs} & russet potatoes \\
	\unit[6]{tbsp} & unsalted butter \\
	\unit[\nicefrac{1}{2}]{cup} & milk \\
	\unit[\nicefrac{1}{2}]{cup} & sour cream\\
	\unit[\nicefrac{1}{2}]{cup} & packed, grated Parmesan\\
				    & freshly ground black pepper\\
	\unit[\nicefrac{3}{4}]{cup} & french lentils\\
	\unit[4]{sprigs} & thyme \\
	\unit[3\nicefrac{1}{2}]{cups} & stock \\
	\unit[1]{tsp} & salt \\
	\unit[4]{tbsp} & unsalted butter \\
	\unit[8]{oz} & mushrooms \\
	1 & large leek \\
	2 & medium carrots \\
	\unit[2]{cloves} & garlic \\
			 & freshly ground black pepper \\
	\unit[2]{tbsp} & tomato paste \\
	\unit[3]{tbsp} & all-purpose flour \\
	\unit[1]{cup} & frozen peas \\
	\unit[1-2]{tsp} & lemon juice \\
}
\preparation{
	\step Boil salted water in a large pot. Add the potatoes and boil for about 15-20 minutes until the potatoes are soft. Alternatively, you can cook the potatoes in an instant pot for 10 minutes on high pressure.
	\step While the potatoes are cooking, prepare the filling. In an oven-safe skillet with high sides bring the lentils, thyme and 2 cups of broth to a simmer with 1 tsp of salt. Reduce the heat and continue to cook, partially covered, until the lentils are tender and most of the liquid is absorbed, about 20 minutes. Transfer to a bowl.
	\step Slice the mushrooms, thinly slice the white part of the leek, dice the carrots, and mince the garlic.
	\step Heat 6 tbsp of the butter and milk together until the butter melts.
	\step Drain the potatoes well and return to the pot, mashing the hot potatoes until smooth. Mix in the butter mixture and sour cream just until blended, then stir in \nicefrac{1}{2} of the parmesan. Season to taste with salt and pepper.
	\step Heat the oven to \unit[375]{\faren}. Melt the remaining butter in the skillet over medium-high heat. Add the mushrooms and cook, stirring frequently until they are a deep golden brown, about 8-10 minutes.
	\step Reduce the heat to medium and add the leek, carrots, and garlic. Continue to cook until tender, about 10 minutes. Season with salt and pepper.
	\step Add the tomato paste and stir, cooking until well combined, about 2-3 minutes.
	\step Sprinkle flour over the mixture, stir and cook for 1 minute. Then add the remaining 1\nicefrac{1}{2} cups of broth, cooked lentils, and peas and cook until thickened. Remove thyme stems and stir in lemon juice to taste.
	\step Top the mixture with dollops of the mashed potatoes, spreading them out to cover the filling. Sprinkle with the remaining parmesan and bake until the filling is bubbling and the topping is golden brown, about 30 minutes. Let stand for about 15 minutes before serving.
}
