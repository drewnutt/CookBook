\begin{recipe}[
	preparationtime={\unit[30]{min}},
	bakingtime={\unit[35]{min}},
	portion=\portion{8},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[435]{\faren}}},
	source=\href{https://www.bonappetit.com/recipe/duchess-baked-potatoes}{Bon Apetit}
	]{Duchess Baked Potatoes}\index{Thanksgiving}
\ingredients[7]{%
	\unit[3 \nicefrac{1}{2}-4]{lb} & yukon gold potatoes \\
	 & salt \\
	5 & egg yolks \\
	\unit[2]{cloves} & garlic\\
	\unit[1 \nicefrac{1}{4}]{cups} & heavy cream\\
	\unit[\nicefrac{3}{4}]{cups} & sour cream\\
	\unit[10]{tbsp} & melted butter\\
	\unit[1]{tsp} & ground pepper \\
}
\preparation{
	\step Boil potatoes over medium-high heat for about 25-35 minutes or until fork tender. This can also be done in the instant pot.
	\step Drain potatoes and let them cool.
	\step Preheat oven to 435\faren.
	\step Whisk egg yolks, pressed garlic, heavy cream, sour cream, melted butter, and ground pepper in a large bowl. Season liberally with salt.
	\step Peel the cooled potatoes and mash them. Add the mashed potatoes to the bowl with the egg mixture.
	\step Fold the potatoes into the egg mixture, careful not to over mix.
	\step Transfer to a 3-quart baking dish (9x13 baking dish) and decorate top with swirls or shingles.
	\step Bake until golden brown and slightly puffed, about 30-40 minutes. Rotate once midway through baking.
}
\suggestion[]{
	\begin{itemize}        
		\item Dish can be prepared up to the last step, one day ahead. Cover and chill, then increase bake time by 5-10 minutes. 
	\end{itemize}        
}
\end{recipe}
