\begin{recipe}[ preparationtime={\unit[1.5]{hours}},
	bakingtime={\unit[30]{min}},
	portion=\portion{8},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[400]{\faren}}},
	source=\href{https://cooking.nytimes.com/recipes/1023638-garlicky-mashed-potato-cake}{NYT Cooking}
	]{Garlicky Mashed Potato Cake}\index{Thanksgiving}

\ingredients[8]{%
	\unit[3]{lb} & Russet potatoes \\
	\unit[6]{tbsp} & butter \\
	\unit[6]{cloves} & garlic \\
			 & salt \\
			 & pepper \\
	pinch & nutmeg \\
	\unit[\nicefrac{1}{2}]{cup} & sour cream\\
	\unit[1]{tbsp} & finely chopped chives \\
}
\preparation{
	\step Bake the potatoes directly on the oven rack for 1 hour and 15 minutes at 425\faren. Alternatively, you can steam the potatoes in an instant pot for 15 minutes and 10 minutes of natural rest.
	\step While cooking the potatoes, smash and peel the garlic.
	\step Lower the oven temperature to 400\faren.
	\step Heat a 10-inch, oven-safe skillet on medium heat. Add the butter and swirl until the butter is melted and foamy.
	\step Add the garlic and stir occasionally until the garlic and butter are golden brown, about 3 minutes. Be careful not to overcook the garlic.
	\step Remove the skillet from the heat and set aside.
	\step Cut the hot potatoes in half and scoop out the flesh onto a cutting board. The skins are no longer needed.
	\step Chop up the flesh to get various sized pieces no larger than a walnut half.
	\step Add the potato to the skillet and season with salt, pepper, and nutmeg.
	\step Add the sour cream and fold gently until the potatoes are evenly coated. Flatten the mixture gently into a single layer.
	\step Cook the mixture over medium-high heat without disturbing it until all of the moisture has evaporated, about 5 to 8 minutes.
	\step Loosen the potato cake in the skillet and transfer to the oven and bake until lightly browned on the surface, about 30-35 minutes.
	\step Remove from the oven and let sit for 10 minutes before flipping onto a serving platter.
}
