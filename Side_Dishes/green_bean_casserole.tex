\begin{recipe}[
	% preparationtime={\unit[]{}},
	bakingtime={\unit[30]{min}},
	portion=\portion{10-12},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[400]{\faren}}},
	source=\href{https://cooking.nytimes.com/recipes/1021599-green-bean-casserole}{NYT Cooking}
	]{Green Bean Casserole}\index{Casserole}\index{Thanksgiving}\index{Cast Iron}
\ingredients[14]{%
	\unit[2 \nicefrac{1}{2}]{lb} & green beans \\
	\unit[1]{lb} & mushrooms\\
	2 & shallots\\
	4 & garlic cloves\\
	\unit[2]{tbsp} & extra-virgin olive oil\\
	 & salt \\
	 & black pepper \\
	\unit[2]{tbsp} & butter \\
	\unit[2]{tbsp} & all-purpose flour \\
	\unit[1]{quart} & heavy cream\\
	\unit[1]{tsp} & cajun seasoning \\
	\unit[\nicefrac{1}{2}]{tsp} & onion powder \\
	\unit[\nicefrac{1}{2}]{tsp} & garlic powder \\
	\unit[\nicefrac{1}{2}]{tsp} & bouillon \\
	\unit[\nicefrac{1}{4}]{tsp} & nutmeg \\
	\unit[1 \nicefrac{3}{4}]{cups} & shredded parmesan \\
	\unit[1]{cup} & bread crumbs \\
	\unit[2]{cups} & fried onions \\
}
\preparation{
	\step Preheat the oven to 400\faren.
	\step Trim the green beans and cut them into 1-inch segments.
	\step Cut the mushrooms into bite-size pieces.
	\step Half the shallots and slice into thin half-moons.
	\step Mince the garlic.
	\step On two sheet pans with a silicone baking mat, add the vegetables and toss with olive oil, salt, and pepper. Spread them evenly on the pan.
	\step Roast until the veggies brown in spots and soften, about 30 minutes. Toss the vegetables halfway through the cooking time.
	\step In a 12-inch cast iron, melt the butter and add the flour. Cook for about 1-2 minutes whisking constantly until it turns a golden color.
	\step Slowly add the heavy cream while whisking. Continue whisking until it starts to thicken, about 5-8 minutes.
	\step Add the cajun seasoning, onion powder, garlic powder, bouillon, nutmeg, and \nicefrac{1}{2} teaspoon black pepper.
	\step Add 1 \nicefrac{1}{2} cups of the of the shredded parmesan and whisk until melted. Add salt as necessary.
	\step When the vegetables are cooked, add them to the saucepan and toss everything until it is warmed and fully mixed.
	\step In a medium bowl, combine the bread crumbs, fried onions, and the remaining \nicefrac{1}{4} cup shredded parmesan. Sprinkle evenly over the casserole.
	\step Set the broiler to high and broil until brown and bubbly, about 1 to 2 minutes.
}
\suggestion[]{
	\begin{itemize}        
		\item Half-and-half can be substituted for the heavy cream.
		\item Instead of a cast-iron, you can use a large saucepan for the cream mixture and transfer to a 9x13 casserole dish after mixing in the vegetables.
	\end{itemize}        
}
\end{recipe}\label{rec:GBCasserole}
