\begin{recipe}[
	preparationtime={\unit[30]{mins}},
	source=\href{https://www.bonappetit.com/recipe/chile-garlic-cucumbers}{Bon Apetit}
	]{Chile Garlic Cucumbers}\index{Cucumber}\index{Garlic}\index{Pickle}
\ingredients[]{%
	\unit[4]{} & Persian cucumbers (or equivalent) \\
		   & kosher salt \\
	\unit[1]{} & serrano chile \\
	\unit[1]{clover} & garlic \\
	\unit[\nicefrac{1}{2}]{cup} & unseasoned rice vinegar \\
	\unit[1]{tbsp} & honey \\
}
\preparation{
	\step Cut the cucumbers in half lengthwise, then cut thinly on a diagonal.
	\step Toss with a large pinch of salt in a medium bowl and let sit ~15 minutes. Toss every 5 minutes or so until cucumbers release their liquid and become wilted.
	\step Meanwhile, thinly slice the chile. In an airtight container large enuogh to fit the cucumbers, combine the chile, garlic, vinegar, honey, and a large pinch of salt. Shake to combine.
	\step Squeeze the cucumbers to remove any excess liquid and transfer to the container with the vinegar mixture. Shake to combine and let sit at least 30 minutes before serving.
	\step Store the cucumbers in the refrigerator for up to 1 week.
}

\end{recipe}
