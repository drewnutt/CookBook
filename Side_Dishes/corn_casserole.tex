\begin{recipe}[bakingtime={\unit[45]{min}},
	portion=\portion{6-8},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[350]{\faren}}},
	source=\href{https://www.pauladeen.com/recipe/corn-casserole/}{Paula Deen}
	]{Corn Casserole}\index{Casserole}\index{Corn}\index{Thanksgiving}
\ingredients[5]{%
	\unit[1]{can} & whole kernel corn \\
	\unit[1]{can} & cream-style corn \\
	\unit[1]{cup} & sour cream\\
	\unit[\nicefrac{1}{2}]{cup} & melted butter \\
	\unit[1]{package} & corn muffin mix \\
	\unit[\nicefrac{1}{2}]{cup} & shredded cheddar cheese \\
}
\preparation{
	\step Preheat the oven to 350\faren.
	\step Drain the whole kernel corn.
	\step Combine all ingredients (except for the cheddar cheese) in a large bowl.
	\step Pour into a greased 9x13 casserole dish.
	\step Bake for 40 to 45 minutes until golden brown.
	\step Cover with cheddar cheese and return to oven for another 5 minutes to melt the cheese.
}
\end{recipe}
