\begin{recipe}[
	preparationtime={\unit[30]{min}},
	bakingtime={\unit[100]{min}},
	portion=\portion{4-6},
	bakingtemperature={\protect\bakingtemperature{fanoven=\unit[450]{\faren}}},
	source=\href{https://www.seriouseats.com/the-best-roast-potatoes-ever-recipe}{Serious Eats}
	]{Crispy Roasted Potatoes}\index{Garlic}\index{Kosher!Milk}\index{Vegan}\index{Sweet Potato}
\ingredients[]{%
	\unit[4]{lbs} & potatoes \\
	\unit[\nicefrac{1}{2}]{tsp} & baking soda \\
	\unit[5]{tbsp} & butter or olive oil \\
	\unit[1]{tsp} & dried rosemary \\
	\unit[3]{cloves} & garlic \\
}
\preparation{
	\step Cut potatoes into 2-3 inch chunks.
	\step Heat 2 quarts of water in a large pot over high heat, until boiling. Add 2 tablespoons of salt, baking soda, and the potatoes. Stir, return to a boil and reduce to a simmer.
	Cook until a knife meets little resistance when inserted into a potato chunk, about 10 minutes after returning to a boil.
	\step Meanwhile combine the oil or butter with rosemary, garlic and a few grinds of black pepper in a small saucepan over medium heat. Cook while constantly stirring and shaking the pan until the garlic just turns golden, about 3 minutes. Strain the mixture through a fine-mesh strainer into a bowl large enough to fit all the potatoes. Keep the solids in a separate bowl.
	\step When the potatoes are cooked, drain carefully and let them rest in the pot for about 30 seconds to allow excess moisture to evaporate. Transfer to the bowl with the infused oil, season to taste with a little more salt and pepper, and toss to coat, shaking the bowl roughly, until a thick layer of mashed potatoâ€“like paste has built up on the potato chunks.
	\step Transfer the potatoes to a large rimmed baking sheet and separate them, spreading them out evenly. Transfer to the oven and roast, without moving, for 20 minutes.
	\step Using a thin, flexible metal spatula to release any stuck potatoes, shake the pan and turn the potatoes. Continue roasting until the potatoes are deep brown and crisp all over, turning and shaking them a few times during cooking, 30 to 40 minutes longer.
	\step Transfer the potatoes to a large bowl and add the reserved garlic and rosemary. Add any other spices or herbs as desired. Toss to coat and season with more salt and pepper to taste.
}
\suggestion[]{
	You can use any type of potato, including a mixture of normal and sweet potatoes.
}
\end{recipe}
